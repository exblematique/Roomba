37 int will__set(struct mosquitto *mosq, const char *topic, int payloadlen, const void *payload, int qos, bool retain)
38 {
39  int rc = MOSQ_ERR_SUCCESS;
40 
41  if(!mosq || !topic) return MOSQ_ERR_INVAL;
42  if(payloadlen < 0 || payloadlen > MQTT_MAX_PAYLOAD) return MOSQ_ERR_PAYLOAD_SIZE;
43  if(payloadlen > 0 && !payload) return MOSQ_ERR_INVAL;
44 
45  if(mosquitto_pub_topic_check(topic)) return MOSQ_ERR_INVAL;
46  if(mosquitto_validate_utf8(topic, strlen(topic))) return MOSQ_ERR_MALFORMED_UTF8;
47 
48  if(mosq->will){
49  mosquitto__free(mosq->will->topic);
50  mosquitto__free(mosq->will->payload);
51  mosquitto__free(mosq->will);

53 
54  mosq->will = mosquitto__calloc(1, sizeof(struct mosquitto_message));
55  if(!mosq->will) return MOSQ_ERR_NOMEM;
56  mosq->will->topic = mosquitto__strdup(topic);
57  if(!mosq->will->topic){
58  rc = MOSQ_ERR_NOMEM;
59  goto cleanup;
60  }
61  mosq->will->payloadlen = payloadlen;
62  if(mosq->will->payloadlen > 0){
63  if(!payload){
64  rc = MOSQ_ERR_INVAL;
65  goto cleanup;
66  }
67  mosq->will->payload = mosquitto__malloc(sizeof(char)*mosq->will->payloadlen);
68  if(!mosq->will->payload){

70  goto cleanup;
71  }
72 
73  memcpy(mosq->will->payload, payload, payloadlen);
74  }
75  mosq->will->qos = qos;
76  mosq->will->retain = retain;
77 
78  return MOSQ_ERR_SUCCESS;
79 
80 cleanup:

82  mosquitto__free(mosq->will->topic);
83  mosquitto__free(mosq->will->payload);
84 
85  mosquitto__free(mosq->will);

87  }
88 
89  return rc;
90 }
