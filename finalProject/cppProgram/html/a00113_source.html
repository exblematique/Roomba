108 /* For PUBACK, PUBCOMP, PUBREC, and PUBREL */
109 int send__command_with_mid(struct mosquitto *mosq, uint8_t command, uint16_t mid, bool dup)
110 {
111  struct mosquitto__packet *packet = NULL;
112  int rc;
113 
114  assert(mosq);
115  packet = mosquitto__calloc(1, sizeof(struct mosquitto__packet));
116  if(!packet) return MOSQ_ERR_NOMEM;
117 
118  packet->command = command;
119  if(dup){
120  packet->command |= 8;
121  }
122  packet->remaining_length = 2;
123  rc = packet__alloc(packet);
124  if(rc){
125  mosquitto__free(packet);
126  return rc;
127  }
128 
129  packet->payload[packet->pos+0] = MOSQ_MSB(mid);
130  packet->payload[packet->pos+1] = MOSQ_LSB(mid);
131 
132  return packet__queue(mosq, packet);
133 }
