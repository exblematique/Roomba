39 int handle__pubrel(struct mosquitto_db *db, struct mosquitto *mosq)
40 {
41  uint16_t mid;
42 #ifndef WITH_BROKER
43  struct mosquitto_message_all *message = NULL;
44 #endif
45  int rc;
46 
47  assert(mosq);
48  if(mosq->protocol == mosq_p_mqtt311){
49  if((mosq->in_packet.command&0x0F) != 0x02){
50  return MOSQ_ERR_PROTOCOL;
51  }
52  }
53  rc = packet__read_uint16(&mosq->in_packet, &mid);
54  if(rc) return rc;
55 #ifdef WITH_BROKER
56  log__printf(NULL, MOSQ_LOG_DEBUG, "Received PUBREL from %s (Mid: %d)", mosq->id, mid);
57 
58  if(db__message_release(db, mosq, mid, mosq_md_in)){
59  /* Message not found. Still send a PUBCOMP anyway because this could be
60  * due to a repeated PUBREL after a client has reconnected. */
61  log__printf(mosq, MOSQ_LOG_WARNING, "Warning: Received PUBREL from %s for an unknown packet identifier %d.", mosq->id, mid);
62  }
63 #else
64  log__printf(mosq, MOSQ_LOG_DEBUG, "Client %s received PUBREL (Mid: %d)", mosq->id, mid);
65 
66  if(!message__remove(mosq, mid, mosq_md_in, &message)){
67  /* Only pass the message on if we have removed it from the queue - this
68  * prevents multiple callbacks for the same message. */
69  pthread_mutex_lock(&mosq->callback_mutex);
70  if(mosq->on_message){
71  mosq->in_callback = true;
72  mosq->on_message(mosq, mosq->userdata, &message->msg);
73  mosq->in_callback = false;
74  }
75  pthread_mutex_unlock(&mosq->callback_mutex);
76  message__cleanup(&message);
77  }
78 #endif
79  rc = send__pubcomp(mosq, mid);
80  if(rc) return rc;
81 
82  return MOSQ_ERR_SUCCESS;
83 }
