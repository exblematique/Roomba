32 int handle__publish(struct mosquitto *mosq)
33 {
34  uint8_t header;
35  struct mosquitto_message_all *message;
36  int rc = 0;
37  uint16_t mid;
38  int slen;
39 
40  assert(mosq);
41 
42  message = mosquitto__calloc(1, sizeof(struct mosquitto_message_all));
43  if(!message) return MOSQ_ERR_NOMEM;
44 
45  header = mosq->in_packet.command;
46 
47  message->dup = (header & 0x08)>>3;
48  message->msg.qos = (header & 0x06)>>1;
49  message->msg.retain = (header & 0x01);
50 
51  rc = packet__read_string(&mosq->in_packet, &message->msg.topic, &slen);
52  if(rc){
53  message__cleanup(&message);
54  return rc;
55  }
56  if(!slen){
57  message__cleanup(&message);
58  return MOSQ_ERR_PROTOCOL;
59  }
60 
61  if(message->msg.qos > 0){
62  rc = packet__read_uint16(&mosq->in_packet, &mid);
63  if(rc){
64  message__cleanup(&message);
65  return rc;
66  }
67  message->msg.mid = (int)mid;
68  }
69 
70  message->msg.payloadlen = mosq->in_packet.remaining_length - mosq->in_packet.pos;
71  if(message->msg.payloadlen){
72  message->msg.payload = mosquitto__calloc(message->msg.payloadlen+1, sizeof(uint8_t));
73  if(!message->msg.payload){
74  message__cleanup(&message);
75  return MOSQ_ERR_NOMEM;
76  }
77  rc = packet__read_bytes(&mosq->in_packet, message->msg.payload, message->msg.payloadlen);
78  if(rc){
79  message__cleanup(&message);
80  return rc;
81  }
82  }
83  log__printf(mosq, MOSQ_LOG_DEBUG,
84  "Client %s received PUBLISH (d%d, q%d, r%d, m%d, '%s', ... (%ld bytes))",
85  mosq->id, message->dup, message->msg.qos, message->msg.retain,
86  message->msg.mid, message->msg.topic,
87  (long)message->msg.payloadlen);
88 
89  message->timestamp = mosquitto_time();
90  switch(message->msg.qos){
91  case 0:
92  pthread_mutex_lock(&mosq->callback_mutex);
93  if(mosq->on_message){
94  mosq->in_callback = true;
95  mosq->on_message(mosq, mosq->userdata, &message->msg);
96  mosq->in_callback = false;
97  }
98  pthread_mutex_unlock(&mosq->callback_mutex);
99  message__cleanup(&message);
100  return MOSQ_ERR_SUCCESS;
101  case 1:
102  rc = send__puback(mosq, message->msg.mid);
103  pthread_mutex_lock(&mosq->callback_mutex);
104  if(mosq->on_message){
105  mosq->in_callback = true;
106  mosq->on_message(mosq, mosq->userdata, &message->msg);
107  mosq->in_callback = false;
108  }
109  pthread_mutex_unlock(&mosq->callback_mutex);
110  message__cleanup(&message);
111  return rc;
112  case 2:
113  rc = send__pubrec(mosq, message->msg.mid);
114  pthread_mutex_lock(&mosq->in_message_mutex);
115  message->state = mosq_ms_wait_for_pubrel;
116  message__queue(mosq, message, mosq_md_in);
117  pthread_mutex_unlock(&mosq->in_message_mutex);
118  return rc;
119  default:
120  message__cleanup(&message);
121  return MOSQ_ERR_PROTOCOL;
122  }
123 }
