66 struct mosquitto *mosquitto_new(const char *id, bool clean_session, void *userdata)
67 {
68  struct mosquitto *mosq = NULL;
69  int rc;
70 
71  if(clean_session == false && id == NULL){
72  errno = EINVAL;
73  return NULL;
74  }
75 
76 #ifndef WIN32
77  signal(SIGPIPE, SIG_IGN);
78 #endif
79 
80  mosq = (struct mosquitto *)mosquitto__calloc(1, sizeof(struct mosquitto));
81  if(mosq){
82  mosq->sock = INVALID_SOCKET;
83  mosq->sockpairR = INVALID_SOCKET;
84  mosq->sockpairW = INVALID_SOCKET;
85 #ifdef WITH_THREADING
86  mosq->thread_id = pthread_self();
87 #endif
88  rc = mosquitto_reinitialise(mosq, id, clean_session, userdata);
89  if(rc){
90  mosquitto_destroy(mosq);
91  if(rc == MOSQ_ERR_INVAL){
92  errno = EINVAL;
93  }else if(rc == MOSQ_ERR_NOMEM){
94  errno = ENOMEM;
95  }
96  return NULL;
97  }
98  }else{
99  errno = ENOMEM;
100  }
101  return mosq;
102 }

113 
114  mosquitto__destroy(mosq);
115  memset(mosq, 0, sizeof(struct mosquitto));
116 
117  if(userdata){
118  mosq->userdata = userdata;
119  }else{

204 void mosquitto__destroy(struct mosquitto *mosq)
205 {
206  struct mosquitto__packet *packet;
207  if(!mosq) return;
208 
209 #ifdef WITH_THREADING
210  if(mosq->threaded == mosq_ts_self && !pthread_equal(mosq->thread_id, pthread_self())){
211  pthread_cancel(mosq->thread_id);
212  pthread_join(mosq->thread_id, NULL);
213  mosq->threaded = mosq_ts_none;
214  }
215 
216  if(mosq->id){
217  /* If mosq->id is not NULL then the client has already been initialised
218  * and so the mutexes need destroying. If mosq->id is NULL, the mutexes
219  * haven't been initialised. */
220  pthread_mutex_destroy(&mosq->callback_mutex);
221  pthread_mutex_destroy(&mosq->log_callback_mutex);
222  pthread_mutex_destroy(&mosq->state_mutex);
223  pthread_mutex_destroy(&mosq->out_packet_mutex);
224  pthread_mutex_destroy(&mosq->current_out_packet_mutex);
225  pthread_mutex_destroy(&mosq->msgtime_mutex);
226  pthread_mutex_destroy(&mosq->in_message_mutex);
227  pthread_mutex_destroy(&mosq->out_message_mutex);
228  pthread_mutex_destroy(&mosq->mid_mutex);
229  }
230 #endif
231  if(mosq->sock != INVALID_SOCKET){
232  net__socket_close(mosq);
233  }
234  message__cleanup_all(mosq);
235  will__clear(mosq);
236 #ifdef WITH_TLS
237  if(mosq->ssl){
238  SSL_free(mosq->ssl);
239  }
240  if(mosq->ssl_ctx){
241  SSL_CTX_free(mosq->ssl_ctx);
242  }
243  mosquitto__free(mosq->tls_cafile);
244  mosquitto__free(mosq->tls_capath);
245  mosquitto__free(mosq->tls_certfile);
246  mosquitto__free(mosq->tls_keyfile);
247  if(mosq->tls_pw_callback) mosq->tls_pw_callback = NULL;
248  mosquitto__free(mosq->tls_version);
249  mosquitto__free(mosq->tls_ciphers);
250  mosquitto__free(mosq->tls_psk);
251  mosquitto__free(mosq->tls_psk_identity);
252 #endif
253 
254  mosquitto__free(mosq->address);
255  mosq->address = NULL;
256 
257  mosquitto__free(mosq->id);
258  mosq->id = NULL;
259 
260  mosquitto__free(mosq->username);
261  mosq->username = NULL;
262 
263  mosquitto__free(mosq->password);
264  mosq->password = NULL;
265 
266  mosquitto__free(mosq->host);
267  mosq->host = NULL;
268 
269  mosquitto__free(mosq->bind_address);
270  mosq->bind_address = NULL;
271 
272  /* Out packet cleanup */
273  if(mosq->out_packet && !mosq->current_out_packet){
274  mosq->current_out_packet = mosq->out_packet;
275  mosq->out_packet = mosq->out_packet->next;
276  }
277  while(mosq->current_out_packet){
278  packet = mosq->current_out_packet;
279  /* Free data and reset values */
280  mosq->current_out_packet = mosq->out_packet;
281  if(mosq->out_packet){
282  mosq->out_packet = mosq->out_packet->next;
283  }
284 
285  packet__cleanup(packet);
286  mosquitto__free(packet);
287  }
288 
289  packet__cleanup(&mosq->in_packet);
290  if(mosq->sockpairR != INVALID_SOCKET){
291  COMPAT_CLOSE(mosq->sockpairR);
292  mosq->sockpairR = INVALID_SOCKET;
293  }
294  if(mosq->sockpairW != INVALID_SOCKET){
295  COMPAT_CLOSE(mosq->sockpairW);
296  mosq->sockpairW = INVALID_SOCKET;
297  }
298 }

400 int mosquitto_sub_topic_tokenise(const char *subtopic, char ***topics, int *count)
401 {
402  int len;
403  int hier_count = 1;
404  int start, stop;
405  int hier;
406  int tlen;
407  int i, j;
408 
409  if(!subtopic || !topics || !count) return MOSQ_ERR_INVAL;
410 
411  len = strlen(subtopic);
412 
413  for(i=0; i<len; i++){
414  if(subtopic[i] == '/'){
415  if(i > len-1){
416  /* Separator at end of line */
417  }else{
418  hier_count++;
419  }
420  }
421  }
422 
423  (*topics) = mosquitto__calloc(hier_count, sizeof(char *));
424  if(!(*topics)) return MOSQ_ERR_NOMEM;
425 
426  start = 0;
427  stop = 0;
428  hier = 0;
429 
430  for(i=0; i<len+1; i++){
431  if(subtopic[i] == '/' || subtopic[i] == '\0'){
432  stop = i;
433  if(start != stop){
434  tlen = stop-start + 1;
435  (*topics)[hier] = mosquitto__calloc(tlen, sizeof(char));
436  if(!(*topics)[hier]){
437  for(j=0; j<hier; j++){
438  mosquitto__free((*topics)[j]);
439  }
440  mosquitto__free((*topics));
441  return MOSQ_ERR_NOMEM;
442  }
443  for(j=start; j<stop; j++){
444  (*topics)[hier][j-start] = subtopic[j];
445  }
446  }
447  start = i+1;
448  hier++;
449  }
450  }
451 
452  *count = hier_count;
453 
454  return MOSQ_ERR_SUCCESS;
455 }
