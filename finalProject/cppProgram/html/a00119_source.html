132 {
133  struct mosquitto__packet *packet = NULL;

136 
137  assert(mosq);

140  packetlen = 2+strlen(topic) + payloadlen;
141  if(qos > 0) packetlen += 2; /* For message id */
142  packet = mosquitto__calloc(1, sizeof(struct mosquitto__packet));
143  if(!packet) return MOSQ_ERR_NOMEM;
144 
145  packet->mid = mid;
146  packet->command = PUBLISH | ((dup&0x1)<<3) | (qos<<1) | retain;
147  packet->remaining_length = packetlen;
148  rc = packet__alloc(packet);
149  if(rc){

151  return rc;
152  }
153  /* Variable header (topic string) */
154  packet__write_string(packet, topic, strlen(topic));
155  if(qos > 0){
156  packet__write_uint16(packet, mid);
157  }
158 
159  /* Payload */
160  if(payloadlen){
161  packet__write_bytes(packet, payload, payloadlen);
162  }
163 
164  return packet__queue(mosq, packet);
