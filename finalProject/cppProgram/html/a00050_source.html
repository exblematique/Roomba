60 static int on_message_simple(struct mosquitto *mosq, void *obj, const struct mosquitto_message *message)
61 {
62  struct userdata__simple *userdata = obj;
63  int rc;
64 
65  if(userdata->max_msg_count == 0){
66  return 0;
67  }
68 
69  /* Don't process stale retained messages if 'want_retained' was false */
70  if(!userdata->want_retained && message->retain){
71  return 0;
72  }
73 
74  userdata->max_msg_count--;
75 
76  rc = mosquitto_message_copy(&userdata->messages[userdata->message_count], message);
77  if(rc){
78  return rc;
79  }
80  userdata->message_count++;
81  if(userdata->max_msg_count == 0){
82  mosquitto_disconnect(mosq);
83  }
84  return 0;
85 }

103 {
104  struct userdata__simple userdata;
105  int rc;
106  int i;
107 
108  if(!topic || msg_count < 1 || !messages){
109  return MOSQ_ERR_INVAL;
110  }
111 
112  *messages = NULL;
113 
114  userdata.messages = calloc(sizeof(struct mosquitto_message), msg_count);
115  if(!userdata.messages){
116  return MOSQ_ERR_NOMEM;
117  }
118  userdata.message_count = 0;
119  userdata.max_msg_count = msg_count;
120  userdata.want_retained = want_retained;
121 
122  rc = mosquitto_subscribe_callback(
123  on_message_simple, &userdata,
124  topic, qos,
125  host, port,
126  client_id, keepalive, clean_session,
127  username, password,
128  will, tls);
129 
130  if(!rc && userdata.max_msg_count == 0){
131  *messages = userdata.messages;
132  return MOSQ_ERR_SUCCESS;
133  }else{
134  for(i=0; i<msg_count; i++){
135  mosquitto_message_free_contents(&userdata.messages[i]);
136  }
137  free(userdata.messages);
138  userdata.messages = NULL;
139  return rc;
140  }
141 }
