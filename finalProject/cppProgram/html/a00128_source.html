60 int mosquitto_socks5_set(struct mosquitto *mosq, const char *host, int port, const char *username, const char *password)
61 {
62 #ifdef WITH_SOCKS
63  if(!mosq) return MOSQ_ERR_INVAL;
64  if(!host || strlen(host) > 256) return MOSQ_ERR_INVAL;
65  if(port < 1 || port > 65535) return MOSQ_ERR_INVAL;
66 
67  mosquitto__free(mosq->socks5_host);
68  mosq->socks5_host = NULL;
69 
70  mosq->socks5_host = mosquitto__strdup(host);
71  if(!mosq->socks5_host){
72  return MOSQ_ERR_NOMEM;
73  }
74 
75  mosq->socks5_port = port;
76 
77  mosquitto__free(mosq->socks5_username);
78  mosq->socks5_username = NULL;
79 
80  mosquitto__free(mosq->socks5_password);
81  mosq->socks5_password = NULL;
82 
83  if(username){
84  mosq->socks5_username = mosquitto__strdup(username);
85  if(!mosq->socks5_username){
86  return MOSQ_ERR_NOMEM;
87  }
88 
89  if(password){
90  mosq->socks5_password = mosquitto__strdup(password);
91  if(!mosq->socks5_password){
92  mosquitto__free(mosq->socks5_username);
93  return MOSQ_ERR_NOMEM;
94  }
95  }
96  }
97 
98  return MOSQ_ERR_SUCCESS;
99 #else
100  return MOSQ_ERR_NOT_SUPPORTED;
101 #endif
102 }

105 int socks5__send(struct mosquitto *mosq)
106 {
107  struct mosquitto__packet *packet;
108  int slen;
109  int ulen, plen;
110 
111  struct in_addr addr_ipv4;
112  struct in6_addr addr_ipv6;
113  int ipv4_pton_result;
114  int ipv6_pton_result;
115 
116  if(mosq->state == mosq_cs_socks5_new){
117  packet = mosquitto__calloc(1, sizeof(struct mosquitto__packet));
118  if(!packet) return MOSQ_ERR_NOMEM;
119 
120  if(mosq->socks5_username){
121  packet->packet_length = 4;
122  }else{
123  packet->packet_length = 3;
124  }
125  packet->payload = mosquitto__malloc(sizeof(uint8_t)*packet->packet_length);
126 
127  packet->payload[0] = 0x05;
128  if(mosq->socks5_username){
129  packet->payload[1] = 2;
130  packet->payload[2] = SOCKS_AUTH_NONE;
131  packet->payload[3] = SOCKS_AUTH_USERPASS;
132  }else{
133  packet->payload[1] = 1;
134  packet->payload[2] = SOCKS_AUTH_NONE;
135  }
136 
137  pthread_mutex_lock(&mosq->state_mutex);
138  mosq->state = mosq_cs_socks5_start;
139  pthread_mutex_unlock(&mosq->state_mutex);
140 
141  mosq->in_packet.pos = 0;
142  mosq->in_packet.packet_length = 2;
143  mosq->in_packet.to_process = 2;
144  mosq->in_packet.payload = mosquitto__malloc(sizeof(uint8_t)*2);
145  if(!mosq->in_packet.payload){
146  mosquitto__free(packet->payload);
147  mosquitto__free(packet);
148  return MOSQ_ERR_NOMEM;
149  }
150 
151  return packet__queue(mosq, packet);
152  }else if(mosq->state == mosq_cs_socks5_auth_ok){
153  packet = mosquitto__calloc(1, sizeof(struct mosquitto__packet));
154  if(!packet) return MOSQ_ERR_NOMEM;
155 
156  ipv4_pton_result = inet_pton(AF_INET, mosq->host, &addr_ipv4);
157  ipv6_pton_result = inet_pton(AF_INET6, mosq->host, &addr_ipv6);
158 
159  if(ipv4_pton_result == 1){
160  packet->packet_length = 10;
161  packet->payload = mosquitto__malloc(sizeof(uint8_t)*packet->packet_length);
162  if(!packet->payload){
163  mosquitto__free(packet);
164  return MOSQ_ERR_NOMEM;
165  }
166  packet->payload[3] = SOCKS_ATYPE_IP_V4;
167  memcpy(&(packet->payload[4]), (const void*)&addr_ipv4, 4);
168  packet->payload[4+4] = MOSQ_MSB(mosq->port);
169  packet->payload[4+4+1] = MOSQ_LSB(mosq->port);
170 
171  }else if(ipv6_pton_result == 1){
172  packet->packet_length = 22;
173  packet->payload = mosquitto__malloc(sizeof(uint8_t)*packet->packet_length);
174  if(!packet->payload){
175  mosquitto__free(packet);
176  return MOSQ_ERR_NOMEM;
177  }
178  packet->payload[3] = SOCKS_ATYPE_IP_V6;
179  memcpy(&(packet->payload[4]), (const void*)&addr_ipv6, 16);
180  packet->payload[4+16] = MOSQ_MSB(mosq->port);
181  packet->payload[4+16+1] = MOSQ_LSB(mosq->port);
182 
183  }else{
184  slen = strlen(mosq->host);
185  if(slen > UCHAR_MAX){
186  return MOSQ_ERR_NOMEM;
187  }
188  packet->packet_length = 7 + slen;
189  packet->payload = mosquitto__malloc(sizeof(uint8_t)*packet->packet_length);
190  if(!packet->payload){
191  mosquitto__free(packet);
192  return MOSQ_ERR_NOMEM;
193  }
194  packet->payload[3] = SOCKS_ATYPE_DOMAINNAME;
195  packet->payload[4] = (uint8_t)slen;
196  memcpy(&(packet->payload[5]), mosq->host, slen);
197  packet->payload[5+slen] = MOSQ_MSB(mosq->port);
198  packet->payload[6+slen] = MOSQ_LSB(mosq->port);
199  }
200  packet->payload[0] = 0x05;
201  packet->payload[1] = 0x01;
202  packet->payload[2] = 0x00;
203 
204  pthread_mutex_lock(&mosq->state_mutex);
205  mosq->state = mosq_cs_socks5_request;
206  pthread_mutex_unlock(&mosq->state_mutex);
207 
208  mosq->in_packet.pos = 0;
209  mosq->in_packet.packet_length = 5;
210  mosq->in_packet.to_process = 5;
211  mosq->in_packet.payload = mosquitto__malloc(sizeof(uint8_t)*5);
212  if(!mosq->in_packet.payload){
213  mosquitto__free(packet->payload);
214  mosquitto__free(packet);
215  return MOSQ_ERR_NOMEM;
216  }
217 
218  return packet__queue(mosq, packet);
219  }else if(mosq->state == mosq_cs_socks5_send_userpass){
220  packet = mosquitto__calloc(1, sizeof(struct mosquitto__packet));
221  if(!packet) return MOSQ_ERR_NOMEM;
222 
223  ulen = strlen(mosq->socks5_username);
224  plen = strlen(mosq->socks5_password);
225  packet->packet_length = 3 + ulen + plen;
226  packet->payload = mosquitto__malloc(sizeof(uint8_t)*packet->packet_length);
227 
228 
229  packet->payload[0] = 0x01;
230  packet->payload[1] = ulen;
231  memcpy(&(packet->payload[2]), mosq->socks5_username, ulen);
232  packet->payload[2+ulen] = plen;
233  memcpy(&(packet->payload[3+ulen]), mosq->socks5_password, plen);
234 
235  pthread_mutex_lock(&mosq->state_mutex);
236  mosq->state = mosq_cs_socks5_userpass_reply;
237  pthread_mutex_unlock(&mosq->state_mutex);
238 
239  mosq->in_packet.pos = 0;
240  mosq->in_packet.packet_length = 2;
241  mosq->in_packet.to_process = 2;
242  mosq->in_packet.payload = mosquitto__malloc(sizeof(uint8_t)*2);
243  if(!mosq->in_packet.payload){
244  mosquitto__free(packet->payload);
245  mosquitto__free(packet);
246  return MOSQ_ERR_NOMEM;
247  }
248 
249  return packet__queue(mosq, packet);
250  }
251  return MOSQ_ERR_SUCCESS;
252 }

254 int socks5__read(struct mosquitto *mosq)
255 {
256  ssize_t len;
257  uint8_t *payload;
258  uint8_t i;
259 
260  if(mosq->state == mosq_cs_socks5_start){
261  while(mosq->in_packet.to_process > 0){
262  len = net__read(mosq, &(mosq->in_packet.payload[mosq->in_packet.pos]), mosq->in_packet.to_process);
263  if(len > 0){
264  mosq->in_packet.pos += len;
265  mosq->in_packet.to_process -= len;
266  }else{
267 #ifdef WIN32
268  errno = WSAGetLastError();
269 #endif
270  if(errno == EAGAIN || errno == COMPAT_EWOULDBLOCK){
271  return MOSQ_ERR_SUCCESS;
272  }else{
273  packet__cleanup(&mosq->in_packet);
274  switch(errno){
275  case 0:
276  return MOSQ_ERR_PROXY;
277  case COMPAT_ECONNRESET:
278  return MOSQ_ERR_CONN_LOST;
279  default:
280  return MOSQ_ERR_ERRNO;
281  }
282  }
283  }
284  }
285  if(mosq->in_packet.payload[0] != 5){
286  packet__cleanup(&mosq->in_packet);
287  return MOSQ_ERR_PROXY;
288  }
289  switch(mosq->in_packet.payload[1]){
290  case SOCKS_AUTH_NONE:
291  packet__cleanup(&mosq->in_packet);
292  mosq->state = mosq_cs_socks5_auth_ok;
293  return socks5__send(mosq);
294  case SOCKS_AUTH_USERPASS:
295  packet__cleanup(&mosq->in_packet);
296  mosq->state = mosq_cs_socks5_send_userpass;
297  return socks5__send(mosq);
298  default:
299  packet__cleanup(&mosq->in_packet);
300  return MOSQ_ERR_AUTH;
301  }
302  }else if(mosq->state == mosq_cs_socks5_userpass_reply){
303  while(mosq->in_packet.to_process > 0){
304  len = net__read(mosq, &(mosq->in_packet.payload[mosq->in_packet.pos]), mosq->in_packet.to_process);
305  if(len > 0){
306  mosq->in_packet.pos += len;
307  mosq->in_packet.to_process -= len;
308  }else{
309 #ifdef WIN32
310  errno = WSAGetLastError();
311 #endif
312  if(errno == EAGAIN || errno == COMPAT_EWOULDBLOCK){
313  return MOSQ_ERR_SUCCESS;
314  }else{
315  packet__cleanup(&mosq->in_packet);
316  switch(errno){
317  case 0:
318  return MOSQ_ERR_PROXY;
319  case COMPAT_ECONNRESET:
320  return MOSQ_ERR_CONN_LOST;
321  default:
322  return MOSQ_ERR_ERRNO;
323  }
324  }
325  }
326  }
327  if(mosq->in_packet.payload[0] != 1){
328  packet__cleanup(&mosq->in_packet);
329  return MOSQ_ERR_PROXY;
330  }
331  if(mosq->in_packet.payload[1] == 0){
332  packet__cleanup(&mosq->in_packet);
333  mosq->state = mosq_cs_socks5_auth_ok;
334  return socks5__send(mosq);
335  }else{
336  i = mosq->in_packet.payload[1];
337  packet__cleanup(&mosq->in_packet);
338  switch(i){
339  case SOCKS_REPLY_CONNECTION_NOT_ALLOWED:
340  return MOSQ_ERR_AUTH;
341 
342  case SOCKS_REPLY_NETWORK_UNREACHABLE:
343  case SOCKS_REPLY_HOST_UNREACHABLE:
344  case SOCKS_REPLY_CONNECTION_REFUSED:
345  return MOSQ_ERR_NO_CONN;
346 
347  case SOCKS_REPLY_GENERAL_FAILURE:
348  case SOCKS_REPLY_TTL_EXPIRED:
349  case SOCKS_REPLY_COMMAND_NOT_SUPPORTED:
350  case SOCKS_REPLY_ADDRESS_TYPE_NOT_SUPPORTED:
351  return MOSQ_ERR_PROXY;
352 
353  default:
354  return MOSQ_ERR_INVAL;
355  }
356  return MOSQ_ERR_PROXY;
357  }
358  }else if(mosq->state == mosq_cs_socks5_request){
359  while(mosq->in_packet.to_process > 0){
360  len = net__read(mosq, &(mosq->in_packet.payload[mosq->in_packet.pos]), mosq->in_packet.to_process);
361  if(len > 0){
362  mosq->in_packet.pos += len;
363  mosq->in_packet.to_process -= len;
364  }else{
365 #ifdef WIN32
366  errno = WSAGetLastError();
367 #endif
368  if(errno == EAGAIN || errno == COMPAT_EWOULDBLOCK){
369  return MOSQ_ERR_SUCCESS;
370  }else{
371  packet__cleanup(&mosq->in_packet);
372  switch(errno){
373  case 0:
374  return MOSQ_ERR_PROXY;
375  case COMPAT_ECONNRESET:
376  return MOSQ_ERR_CONN_LOST;
377  default:
378  return MOSQ_ERR_ERRNO;
379  }
380  }
381  }
382  }
383 
384  if(mosq->in_packet.packet_length == 5){
385  /* First part of the packet has been received, we now know what else to expect. */
386  if(mosq->in_packet.payload[3] == SOCKS_ATYPE_IP_V4){
387  mosq->in_packet.to_process += 4+2-1; /* 4 bytes IPv4, 2 bytes port, -1 byte because we've already read the first byte */
388  mosq->in_packet.packet_length += 4+2-1;
389  }else if(mosq->in_packet.payload[3] == SOCKS_ATYPE_IP_V6){
390  mosq->in_packet.to_process += 16+2-1; /* 16 bytes IPv6, 2 bytes port, -1 byte because we've already read the first byte */
391  mosq->in_packet.packet_length += 16+2-1;
392  }else if(mosq->in_packet.payload[3] == SOCKS_ATYPE_DOMAINNAME){
393  if(mosq->in_packet.payload[4] > 0 && mosq->in_packet.payload[4] <= 255){
394  mosq->in_packet.to_process += mosq->in_packet.payload[4];
395  mosq->in_packet.packet_length += mosq->in_packet.payload[4];
396  }
397  }else{
398  packet__cleanup(&mosq->in_packet);
399  return MOSQ_ERR_PROTOCOL;
400  }
401  payload = mosquitto__realloc(mosq->in_packet.payload, mosq->in_packet.packet_length);
402  if(payload){
403  mosq->in_packet.payload = payload;
404  }else{
405  packet__cleanup(&mosq->in_packet);
406  return MOSQ_ERR_NOMEM;
407  }
408  payload = mosquitto__realloc(mosq->in_packet.payload, mosq->in_packet.packet_length);
409  if(payload){
410  mosq->in_packet.payload = payload;
411  }else{
412  packet__cleanup(&mosq->in_packet);
413  return MOSQ_ERR_NOMEM;
414  }
415  return MOSQ_ERR_SUCCESS;
416  }
417 
418  /* Entire packet is now read. */
419  if(mosq->in_packet.payload[0] != 5){
420  packet__cleanup(&mosq->in_packet);
421  return MOSQ_ERR_PROXY;
422  }
423  if(mosq->in_packet.payload[1] == 0){
424  /* Auth passed */
425  packet__cleanup(&mosq->in_packet);
426  mosq->state = mosq_cs_new;
427  return send__connect(mosq, mosq->keepalive, mosq->clean_session);
428  }else{
429  i = mosq->in_packet.payload[1];
430  packet__cleanup(&mosq->in_packet);
431  mosq->state = mosq_cs_socks5_new;
432  switch(i){
433  case SOCKS_REPLY_CONNECTION_NOT_ALLOWED:
434  return MOSQ_ERR_AUTH;
435 
436  case SOCKS_REPLY_NETWORK_UNREACHABLE:
437  case SOCKS_REPLY_HOST_UNREACHABLE:
438  case SOCKS_REPLY_CONNECTION_REFUSED:
439  return MOSQ_ERR_NO_CONN;
440 
441  case SOCKS_REPLY_GENERAL_FAILURE:
442  case SOCKS_REPLY_TTL_EXPIRED:
443  case SOCKS_REPLY_COMMAND_NOT_SUPPORTED:
444  case SOCKS_REPLY_ADDRESS_TYPE_NOT_SUPPORTED:
445  return MOSQ_ERR_PROXY;
446 
447  default:
448  return MOSQ_ERR_INVAL;
449  }
450  }
451  }else{
452  return packet__read(mosq);
453  }
454  return MOSQ_ERR_SUCCESS;
455 }
