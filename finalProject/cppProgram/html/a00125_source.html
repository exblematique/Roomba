35 {
36  /* FIXME - only deals with a single topic */

42  assert(mosq);

45  packet = mosquitto__calloc(1, sizeof(struct mosquitto__packet));
46  if(!packet) return MOSQ_ERR_NOMEM;
47 
48  packetlen = 2 + 2+strlen(topic);
49 
50  packet->command = UNSUBSCRIBE | (1<<1);
51  packet->remaining_length = packetlen;

57

62

65 
66 #ifdef WITH_BROKER

69 # endif
70 #else
71  log__printf(mosq, MOSQ_LOG_DEBUG, "Client %s sending UNSUBSCRIBE (Mid: %d, Topic: %s)", mosq->id, local_mid, topic);
