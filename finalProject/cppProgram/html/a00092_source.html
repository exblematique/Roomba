255 int mosquitto_opts_set(struct mosquitto *mosq, enum mosq_opt_t option, void *value)
256 {
257  int ival;
258 
259  if(!mosq || !value) return MOSQ_ERR_INVAL;
260 
261  switch(option){
262  case MOSQ_OPT_PROTOCOL_VERSION:
263  ival = *((int *)value);
264  if(ival == MQTT_PROTOCOL_V31){
265  mosq->protocol = mosq_p_mqtt31;
266  }else if(ival == MQTT_PROTOCOL_V311){
267  mosq->protocol = mosq_p_mqtt311;
268  }else{
269  return MOSQ_ERR_INVAL;
270  }
271  break;
272  case MOSQ_OPT_SSL_CTX:
273 #ifdef WITH_TLS
274  mosq->ssl_ctx = (SSL_CTX *)value;
275  if(mosq->ssl_ctx){
276 #if (OPENSSL_VERSION_NUMBER >= 0x10100000L) && !defined(LIBRESSL_VERSION_NUMBER)
277  SSL_CTX_up_ref(mosq->ssl_ctx);
278 #else
279  CRYPTO_add(&(mosq->ssl_ctx)->references, 1, CRYPTO_LOCK_SSL_CTX);
280 #endif
281  }
282  break;
283 #else
284  return MOSQ_ERR_NOT_SUPPORTED;
285 #endif
286  case MOSQ_OPT_SSL_CTX_WITH_DEFAULTS:
287 #if defined(WITH_TLS) && OPENSSL_VERSION_NUMBER >= 0x10100000L
288  mosq->ssl_ctx_defaults = true;
289  break;
290 #else
291  return MOSQ_ERR_NOT_SUPPORTED;
292 #endif
293  default:
294  return MOSQ_ERR_INVAL;
295  }
296  return MOSQ_ERR_SUCCESS;
297 }
