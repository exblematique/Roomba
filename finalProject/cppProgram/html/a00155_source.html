49 int mosquitto__check_keepalive(struct mosquitto_db *db, struct mosquitto *mosq)
50 #else
51 int mosquitto__check_keepalive(struct mosquitto *mosq)
52 #endif
53 {
54  time_t next_msg_out;
55  time_t last_msg_in;
56  time_t now = mosquitto_time();
57 #ifndef WITH_BROKER
58  int rc;
59 #endif
60 
61  assert(mosq);
62 #if defined(WITH_BROKER) && defined(WITH_BRIDGE)
63  /* Check if a lazy bridge should be timed out due to idle. */
64  if(mosq->bridge && mosq->bridge->start_type == bst_lazy
65  && mosq->sock != INVALID_SOCKET
66  && now - mosq->next_msg_out - mosq->keepalive >= mosq->bridge->idle_timeout){
67 
68  log__printf(NULL, MOSQ_LOG_NOTICE, "Bridge connection %s has exceeded idle timeout, disconnecting.", mosq->id);
69  net__socket_close(db, mosq);
70  return MOSQ_ERR_SUCCESS;
71  }
72 #endif
73  pthread_mutex_lock(&mosq->msgtime_mutex);
74  next_msg_out = mosq->next_msg_out;
75  last_msg_in = mosq->last_msg_in;
76  pthread_mutex_unlock(&mosq->msgtime_mutex);
77  if(mosq->keepalive && mosq->sock != INVALID_SOCKET &&
78  (now >= next_msg_out || now - last_msg_in >= mosq->keepalive)){
79 
80  if(mosq->state == mosq_cs_connected && mosq->ping_t == 0){
81  send__pingreq(mosq);
82  /* Reset last msg times to give the server time to send a pingresp */
83  pthread_mutex_lock(&mosq->msgtime_mutex);
84  mosq->last_msg_in = now;
85  mosq->next_msg_out = now + mosq->keepalive;
86  pthread_mutex_unlock(&mosq->msgtime_mutex);
87  }else{
88 #ifdef WITH_BROKER
89  if(mosq->listener){
90  mosq->listener->client_count--;
91  assert(mosq->listener->client_count >= 0);
92  }
93  mosq->listener = NULL;
94  net__socket_close(db, mosq);
95 #else
96  net__socket_close(mosq);
97  pthread_mutex_lock(&mosq->state_mutex);
98  if(mosq->state == mosq_cs_disconnecting){
99  rc = MOSQ_ERR_SUCCESS;
100  }else{
101  rc = MOSQ_ERR_KEEPALIVE;
102  }
103  pthread_mutex_unlock(&mosq->state_mutex);
104  pthread_mutex_lock(&mosq->callback_mutex);
105  if(mosq->on_disconnect){
106  mosq->in_callback = true;
107  mosq->on_disconnect(mosq, mosq->userdata, rc);
108  mosq->in_callback = false;
109  }
110  pthread_mutex_unlock(&mosq->callback_mutex);
111 
112  return rc;
113 #endif
114  }
115  }
116  return MOSQ_ERR_SUCCESS;
117 }

175 /* Check that a topic used for subscriptions is valid.
176  * Search for + or # in a topic, check they aren't in invalid positions such as
177  * foo/#/bar, foo/+bar or foo/bar#.
178  * Return MOSQ_ERR_INVAL if invalid position found.
179  * Also returns MOSQ_ERR_INVAL if the topic string is too long.
180  * Returns MOSQ_ERR_SUCCESS if everything is fine.
181  */
182 int mosquitto_sub_topic_check(const char *str)
183 {
184  char c = '\0';
185  int len = 0;
186  while(str && str[0]){
187  if(str[0] == '+'){
188  if((c != '\0' && c != '/') || (str[1] != '\0' && str[1] != '/')){
189  return MOSQ_ERR_INVAL;
190  }
191  }else if(str[0] == '#'){
192  if((c != '\0' && c != '/') || str[1] != '\0'){
193  return MOSQ_ERR_INVAL;
194  }
195  }
196  len++;
197  c = str[0];
198  str = &str[1];
199  }
200  if(len > 65535) return MOSQ_ERR_INVAL;
201 
202  return MOSQ_ERR_SUCCESS;
203 }

245 /* Does a topic match a subscription? */
246 int mosquitto_topic_matches_sub2(const char *sub, size_t sublen, const char *topic, size_t topiclen, bool *result)
247 {
248  int spos, tpos;
249  bool multilevel_wildcard = false;
250 
251  if(!result) return MOSQ_ERR_INVAL;
252  *result = false;
253 
254  if(!sub || !topic){
255  return MOSQ_ERR_INVAL;
256  }
257 
258  if(!sublen || !topiclen){
259  *result = false;
260  return MOSQ_ERR_INVAL;
261  }
262 
263  if(sublen && topiclen){
264  if((sub[0] == '$' && topic[0] != '$')
265  || (topic[0] == '$' && sub[0] != '$')){
266 
267  return MOSQ_ERR_SUCCESS;
268  }
269  }
270 
271  spos = 0;
272  tpos = 0;
273 
274  while(spos < sublen && tpos <= topiclen){
275  if(sub[spos] == topic[tpos]){
276  if(tpos == topiclen-1){
277  /* Check for e.g. foo matching foo/# */
278  if(spos == sublen-3
279  && sub[spos+1] == '/'
280  && sub[spos+2] == '#'){
281  *result = true;
282  multilevel_wildcard = true;
283  return MOSQ_ERR_SUCCESS;
284  }
285  }
286  spos++;
287  tpos++;
288  if(spos == sublen && tpos == topiclen){
289  *result = true;
290  return MOSQ_ERR_SUCCESS;
291  }else if(tpos == topiclen && spos == sublen-1 && sub[spos] == '+'){
292  if(spos > 0 && sub[spos-1] != '/'){
293  return MOSQ_ERR_INVAL;
294  }
295  spos++;
296  *result = true;
297  return MOSQ_ERR_SUCCESS;
298  }
299  }else{
300  if(sub[spos] == '+'){
301  /* Check for bad "+foo" or "a/+foo" subscription */
302  if(spos > 0 && sub[spos-1] != '/'){
303  return MOSQ_ERR_INVAL;
304  }
305  /* Check for bad "foo+" or "foo+/a" subscription */
306  if(spos < sublen-1 && sub[spos+1] != '/'){
307  return MOSQ_ERR_INVAL;
308  }
309  spos++;
310  while(tpos < topiclen && topic[tpos] != '/'){
311  tpos++;
312  }
313  if(tpos == topiclen && spos == sublen){
314  *result = true;
315  return MOSQ_ERR_SUCCESS;
316  }
317  }else if(sub[spos] == '#'){
318  if(spos > 0 && sub[spos-1] != '/'){
319  return MOSQ_ERR_INVAL;
320  }
321  multilevel_wildcard = true;
322  if(spos+1 != sublen){
323  return MOSQ_ERR_INVAL;
324  }else{
325  *result = true;
326  return MOSQ_ERR_SUCCESS;
327  }
328  }else{
329  /* Check for e.g. foo/bar matching foo/+/# */
330  if(spos > 0
331  && spos+2 == sublen
332  && tpos == topiclen
333  && sub[spos-1] == '+'
334  && sub[spos] == '/'
335  && sub[spos+1] == '#')
336  {
337  *result = true;
338  multilevel_wildcard = true;
339  return MOSQ_ERR_SUCCESS;
340  }
341  return MOSQ_ERR_SUCCESS;
342  }
343  }
344  }
345  if(multilevel_wildcard == false && (tpos < topiclen || spos < sublen)){
346  *result = false;
347  }
348 
349  return MOSQ_ERR_SUCCESS;
350 }

388 {
389 #ifdef WIN32

398  SECURITY_ATTRIBUTES sec;

403  SECURITY_DESCRIPTOR sd;
404

407  return NULL;

419  sec.bInheritHandle = FALSE;

421
