61 int mosquitto_message_copy(struct mosquitto_message *dst, const struct mosquitto_message *src)
62 {
63  if(!dst || !src) return MOSQ_ERR_INVAL;
64 
65  dst->mid = src->mid;
66  dst->topic = mosquitto__strdup(src->topic);
67  if(!dst->topic) return MOSQ_ERR_NOMEM;
68  dst->qos = src->qos;
69  dst->retain = src->retain;
70  if(src->payloadlen){
71  dst->payload = mosquitto__calloc(src->payloadlen+1, sizeof(uint8_t));
72  if(!dst->payload){
73  mosquitto__free(dst->topic);
74  return MOSQ_ERR_NOMEM;
75  }
76  memcpy(dst->payload, src->payload, src->payloadlen);
77  dst->payloadlen = src->payloadlen;
78  }else{
79  dst->payloadlen = 0;
80  dst->payload = NULL;
81  }
82  return MOSQ_ERR_SUCCESS;
83 }

119 int message__queue(struct mosquitto *mosq, struct mosquitto_message_all *message, enum mosquitto_msg_direction dir)
120 {
121  int rc = 0;
122 
123  /* mosq->*_message_mutex should be locked before entering this function */
124  assert(mosq);
125  assert(message);
126 
127  if(dir == mosq_md_out){
128  mosq->out_queue_len++;
129  message->next = NULL;
130  if(mosq->out_messages_last){
131  mosq->out_messages_last->next = message;
132  }else{
133  mosq->out_messages = message;
134  }
135  mosq->out_messages_last = message;
136  if(message->msg.qos > 0){
137  if(mosq->max_inflight_messages == 0 || mosq->inflight_messages < mosq->max_inflight_messages){
138  mosq->inflight_messages++;
139  }else{
140  rc = 1;
141  }
142  }
143  }else{
144  mosq->in_queue_len++;
145  message->next = NULL;
146  if(mosq->in_messages_last){
147  mosq->in_messages_last->next = message;
148  }else{
149  mosq->in_messages = message;
150  }
151  mosq->in_messages_last = message;
152  }
153  return rc;
154 }

156 void message__reconnect_reset(struct mosquitto *mosq)
157 {
158  struct mosquitto_message_all *message;
159  struct mosquitto_message_all *prev = NULL;
160  assert(mosq);
161 
162  pthread_mutex_lock(&mosq->in_message_mutex);
163  message = mosq->in_messages;
164  mosq->in_queue_len = 0;
165  while(message){
166  mosq->in_queue_len++;
167  message->timestamp = 0;
168  if(message->msg.qos != 2){
169  if(prev){
170  prev->next = message->next;
171  message__cleanup(&message);
172  message = prev;
173  }else{
174  mosq->in_messages = message->next;
175  message__cleanup(&message);
176  message = mosq->in_messages;
177  }
178  }else{
179  /* Message state can be preserved here because it should match
180  * whatever the client has got. */
181  }
182  prev = message;
183  if(message) message = message->next;
184  }
185  mosq->in_messages_last = prev;
186  pthread_mutex_unlock(&mosq->in_message_mutex);
187 
188 
189  pthread_mutex_lock(&mosq->out_message_mutex);
190  mosq->inflight_messages = 0;
191  message = mosq->out_messages;
192  mosq->out_queue_len = 0;
193  while(message){
194  mosq->out_queue_len++;
195  message->timestamp = 0;
196 
197  if(mosq->max_inflight_messages == 0 || mosq->inflight_messages < mosq->max_inflight_messages){
198  if(message->msg.qos > 0){
199  mosq->inflight_messages++;
200  }
201  if(message->msg.qos == 1){
202  message->state = mosq_ms_publish_qos1;
203  }else if(message->msg.qos == 2){
204  if(message->state == mosq_ms_wait_for_pubrec){
205  message->state = mosq_ms_publish_qos2;
206  }else if(message->state == mosq_ms_wait_for_pubcomp){
207  message->state = mosq_ms_resend_pubrel;
208  }
209  /* Should be able to preserve state. */
210  }
211  }else{
212  message->state = mosq_ms_invalid;
213  }
214  prev = message;
215  message = message->next;
216  }
217  mosq->out_messages_last = prev;
218  pthread_mutex_unlock(&mosq->out_message_mutex);
219 }

221 int message__remove(struct mosquitto *mosq, uint16_t mid, enum mosquitto_msg_direction dir, struct mosquitto_message_all **message)
222 {
223  struct mosquitto_message_all *cur, *prev = NULL;
224  bool found = false;
225  int rc;
226  assert(mosq);
227  assert(message);
228 
229  if(dir == mosq_md_out){
230  pthread_mutex_lock(&mosq->out_message_mutex);
231  cur = mosq->out_messages;
232  while(cur){
233  if(cur->msg.mid == mid){
234  if(prev){
235  prev->next = cur->next;
236  }else{
237  mosq->out_messages = cur->next;
238  }
239  *message = cur;
240  mosq->out_queue_len--;
241  if(cur->next == NULL){
242  mosq->out_messages_last = prev;
243  }else if(!mosq->out_messages){
244  mosq->out_messages_last = NULL;
245  }
246  if(cur->msg.qos > 0){
247  mosq->inflight_messages--;
248  }
249  found = true;
250  break;
251  }
252  prev = cur;
253  cur = cur->next;
254  }
255 
256  if(found){
257  cur = mosq->out_messages;
258  while(cur){
259  if(mosq->max_inflight_messages == 0 || mosq->inflight_messages < mosq->max_inflight_messages){
260  if(cur->msg.qos > 0 && cur->state == mosq_ms_invalid){
261  mosq->inflight_messages++;
262  if(cur->msg.qos == 1){
263  cur->state = mosq_ms_wait_for_puback;
264  }else if(cur->msg.qos == 2){
265  cur->state = mosq_ms_wait_for_pubrec;
266  }
267  rc = send__publish(mosq, cur->msg.mid, cur->msg.topic, cur->msg.payloadlen, cur->msg.payload, cur->msg.qos, cur->msg.retain, cur->dup);
268  if(rc){
269  pthread_mutex_unlock(&mosq->out_message_mutex);
270  return rc;
271  }
272  }
273  }else{
274  pthread_mutex_unlock(&mosq->out_message_mutex);
275  return MOSQ_ERR_SUCCESS;
276  }
277  cur = cur->next;
278  }
279  pthread_mutex_unlock(&mosq->out_message_mutex);
280  return MOSQ_ERR_SUCCESS;
281  }else{
282  pthread_mutex_unlock(&mosq->out_message_mutex);
283  return MOSQ_ERR_NOT_FOUND;
284  }
285  }else{
286  pthread_mutex_lock(&mosq->in_message_mutex);
287  cur = mosq->in_messages;
288  while(cur){
289  if(cur->msg.mid == mid){
290  if(prev){
291  prev->next = cur->next;
292  }else{
293  mosq->in_messages = cur->next;
294  }
295  *message = cur;
296  mosq->in_queue_len--;
297  if(cur->next == NULL){
298  mosq->in_messages_last = prev;
299  }else if(!mosq->in_messages){
300  mosq->in_messages_last = NULL;
301  }
302  found = true;
303  break;
304  }
305  prev = cur;
306  cur = cur->next;
307  }
308 
309  pthread_mutex_unlock(&mosq->in_message_mutex);
310  if(found){
311  return MOSQ_ERR_SUCCESS;
312  }else{
313  return MOSQ_ERR_NOT_FOUND;
314  }
315  }
316 }

319 void message__retry_check_actual(struct mosquitto *mosq, struct mosquitto_message_all *messages, pthread_mutex_t *mutex)
320 #else
321 void message__retry_check_actual(struct mosquitto *mosq, struct mosquitto_message_all *messages)
322 #endif
323 {
324  time_t now = mosquitto_time();
325  assert(mosq);
326 
327 #ifdef WITH_THREADING
328  pthread_mutex_lock(mutex);
329 #endif
330 
331  while(messages){
332  switch(messages->state){
333  case mosq_ms_publish_qos1:
334  case mosq_ms_publish_qos2:
335  messages->timestamp = now;
336  messages->dup = true;
337  send__publish(mosq, messages->msg.mid, messages->msg.topic, messages->msg.payloadlen, messages->msg.payload, messages->msg.qos, messages->msg.retain, messages->dup);
338  break;
339  case mosq_ms_wait_for_pubrel:
340  messages->timestamp = now;
341  messages->dup = true;
342  send__pubrec(mosq, messages->msg.mid);
343  break;
344  case mosq_ms_resend_pubrel:
345  case mosq_ms_wait_for_pubcomp:
346  messages->timestamp = now;
347  messages->dup = true;
348  send__pubrel(mosq, messages->msg.mid);
349  break;
350  default:
351  break;
352  }
353  messages = messages->next;
354  }
355 #ifdef WITH_THREADING
356  pthread_mutex_unlock(mutex);
357 #endif
358 }
